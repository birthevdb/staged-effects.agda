module _ where

  NatExpr : Con

  ⟦nat⟧ :   ⦃ NatOpSig V ⊏ ζ ⦄
          → ⦃ ℕ ⊂ V ⦄
            ------------------
          → NatExpr ⟨ ζ ⟩⇒ V


  denote ⟦nat⟧ (n , _) = nat n

  nat' : ⦃ NatExpr ≺ K ⦄ → ℕ → μ K

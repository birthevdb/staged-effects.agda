open import Function
open import Data.Product
open import Data.Sum

module Container where

module _ where

  record Con : Set₁ where
    constructor _▹_ 
    field S : Set
          P : S → Set 

  open Con public

  ⟦_⟧ᶜ : Con → Set → Set
  ⟦ S ▹ P ⟧ᶜ X = ∃ λ s → P s → X

  data μ (K : Con) : Set where
    ⟨_⟩ : ⟦ K ⟧ᶜ (μ K) → μ K

  variable
    K K₁ K₂ K₃ : Con
    V V₁ V₂ V₃ : Set
    A B : Set

  mapᶜ : (A → B) → ⟦ K ⟧ᶜ A → ⟦ K ⟧ᶜ B 
  mapᶜ f (s , p) = s , f ∘ p

  _⇒_ : (K : Con) → (V : Set) → Set
  K ⇒ V = ⟦ K ⟧ᶜ V → V

  infix 10 _⇒_

  foldᶜ : (K ⇒ A) → μ K → A
  foldᶜ f ⟨ s , p ⟩ = f (s , foldᶜ f ∘ p)

  _∪_ : (C₁ C₂ : Con) → Con
  S (C₁ ∪ C₂) = S C₁ ⊎ S C₂
  P (C₁ ∪ C₂) (inj₁ x) = P C₁ x
  P (C₁ ∪ C₂) (inj₂ y) = P C₂ y

  infixr 15 _∪_

  record _≺_ (K₁ K₂ : Con) : Set₁ where
    field inj : ⟦ K₁ ⟧ᶜ A → ⟦ K₂ ⟧ᶜ A

  open _≺_  ⦃...⦄

  instance ≺-refl : K ≺ K
  _≺_.inj ≺-refl = id

  instance ≺-left : K₁ ≺ (K₁ ∪ K₂)
  _≺_.inj ≺-left (s , p) = inj₁ s , p

  instance ≺-right : ⦃ K₁ ≺ K₃ ⦄ → K₁ ≺ (K₂ ∪ K₃)
  _≺_.inj ≺-right x with inj x
  ... | s , p = inj₂ s , p

  inject : ⦃ K₁ ≺ K₂ ⦄ → ⟦ K₁ ⟧ᶜ (μ K₂) → μ K₂
  inject x = ⟨ inj x ⟩

module _ where 

  infixr 15 _⊙_
  _⊙_ : (K₁ ⇒ V) → (K₂ ⇒ V) → (K₁ ∪ K₂ ⇒ V)
  (f ⊙ g) (inj₁ x , p) = f (x , p)
  (f ⊙ g) (inj₂ y , p) = g (y , p)
